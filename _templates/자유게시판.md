const authorMap = {
  "김범진": "김범진(PM)", "beomjinkim2000": "김범진(PM)",
  "박창준": "박창준(Exp)", "cjkj1234": "박창준(Exp)",

};

// _config/me.md 에서 저장된 ID 읽기
const configPath = "_config/me.md";
let gitUser = "";
try {
  const f = app.vault.getAbstractFileByPath(configPath);
  if (f) gitUser = (await app.vault.read(f)).trim();
} catch(e) {}

// 없으면 git config 시도
if (!gitUser) {
  try {
    gitUser = (await tp.system.exec_command("git config user.name")).trim();
  } catch(e) {}
}

// 그래도 없으면 한 번만 프롬프트 → 저장
if (!gitUser) {
  gitUser = await tp.system.prompt("GitHub ID를 입력하세요 (한 번만 저장됩니다)");
  try {
    const folder = app.vault.getAbstractFileByPath("_config");
    if (!folder) await app.vault.createFolder("_config");
    await app.vault.create(configPath, gitUser);
  } catch(e) {}
}

const author = authorMap[gitUser] || gitUser;
